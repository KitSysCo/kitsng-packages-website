Quick Start Guide

Get up and running with kitsng-form-factory in minutes!

Basic Example

import { Component } from '@angular/core';
import { FormGroup } from '@angular/forms';
import { KitsngFormFactoryModel } from 'kitsng-form-factory';


@Component({
  selector: 'app-example',
  template: 
    
    
  
})
export class ExampleComponent {
  form = new FormGroup({});



fields: KitsngFormFactoryModel[] = [
    {
      controlType: 'input',
      colSize: 'col-span-12',
      options: {
        formControlName: 'firstName',
        label: 'First Name',
        placeholder: 'Enter first name',
        validators: {
          required: true,
          minLength: 2
        }
      }
    },
    {
      controlType: 'input',
      colSize: 'col-span-12',
      options: {
        formControlName: 'lastName',
        label: 'Last Name',
        placeholder: 'Enter last name',
        validators: {
          required: true
        }
      }
    },
    {
      controlType: 'dropdown',
      colSize: 'col-span-12',
      options: {
        formControlName: 'country',
        label: 'Country',
        placeholder: 'Select a country',
        dropdownOptions: [
          { id: 1, text: 'United States' },
          { id: 2, text: 'Canada' },
          { id: 3, text: 'United Kingdom' }
        ],
        optionLabel: 'text',
        optionValue: 'id',
        validators: {
          required: true
        }
      }
    },
    {
      controlType: 'textarea',
      colSize: 'col-span-12',
      options: {
        formControlName: 'bio',
        label: 'Biography',
        placeholder: 'Tell us about yourself',
        rows: 5,
        validators: {
          maxLength: 500
        }
      }
    }
  ];



onSubmit() {
    if (this.form.valid) {
      console.log('Form is valid!', this.form.value);
      // Submit your form data
    } else {
      console.log('Form has errors');
      this.form.markAllAsTouched();
    }
  }
}

Understanding the Structure

Field Configuration

Each field in the fields array has:

  1. controlType - The type of form field (e.g., 'input', 'dropdown')
  2. colSize - Tailwind CSS class for column sizing (e.g., 'col-span-12', 'col-span-6')
  3. options - Field-specific configuration

Options Object

The options object contains:

  • formControlName - Name of the form control (required)
  • label - Label displayed above the field
  • placeholder - Placeholder text
  • validators - Validation rules
  • Field-specific options - See individual field documentation

Common Patterns

Form Submission

onSubmit() {
  if (this.form.valid) {
    // Handle valid form
    const formData = this.form.value;
    console.log(formData);
  } else {
    // Handle invalid form
    this.form.markAllAsTouched();
  }
}

Getting Form Values

// Get all form values
const allValues = this.form.value;


// Get specific field value
const firstName = this.form.get('firstName')?.value;



// Get form as JSON
const formJson = JSON.stringify(this.form.value);

Setting Form Values

// Set all values at once
this.form.patchValue({
  firstName: 'John',
  lastName: 'Doe',
  country: 1
});


// Set single value
this.form.get('firstName')?.setValue('John');

Listening to Field Changes

// In component template




// In component class
onFieldChange(event: { fieldName: string; value: any; form: FormGroup }) {
  console.log(Field ${event.fieldName} changed to:, event.value);
  
  // Example: Update dependent field
  if (event.fieldName === 'country') {
    this.updateCities(event.value);
  }
}

Advanced Features

Conditional Visibility

Fields can be shown or hidden based on other field values:

{
  controlType: 'dropdown',
  colSize: 'col-span-12',
  options: {
    formControlName: 'country',
    label: 'Country',
    dropdownOptions: [
      { id: 1, text: 'USA' },
      { id: 2, text: 'Canada' }
    ]
  }
},
{
  controlType: 'input',
  colSize: 'col-span-12',
  options: {
    formControlName: 'state',
    label: 'State',
    // Only show when country is USA
    visibleWhen: {
      field: 'country',
      value: 1,
      operator: 'equals'
    }
  }
}

Custom and Async Validators

{
  controlType: 'input',
  options: {
    formControlName: 'username',
    label: 'Username',
    validators: {
      required: true,
      minLength: 3,
      // Custom validator
      custom: (control) => {
        if (control.value && control.value.includes('admin')) {
          return { forbidden: true };
        }
        return null;
      },
      // Async validator
      async: (control) => {
        return this.userService.checkUsername(control.value).pipe(
          map(exists => exists ? { usernameTaken: true } : null)
        );
      }
    }
  }
}

Control Layout Options

Control how labels and controls are positioned:

{
  controlType: 'input',
  options: {
    formControlName: 'email',
    label: 'Email',
    // Vertical layout (default) - label above control
    controlLayout: 'vertical'
  }
},
{
  controlType: 'dropdown',
  options: {
    formControlName: 'country',
    label: 'Country',
    // Horizontal layout - label and control side-by-side
    controlLayout: 'horizontal',
    labelColSize: 'col-span-3',  // Label takes 3 columns
    controlColSize: 'col-span-9' // Control takes 9 columns
  }
}

Dynamic Options

Options can be computed based on other field values:

{
  controlType: 'dropdown',
  options: {
    formControlName: 'city',
    label: 'City',
    // Options depend on country selection
    dynamicOptions: (form) => {
      const country = form.get('country')?.value;
      if (country === 1) {
        return [
          { id: 1, text: 'New York' },
          { id: 2, text: 'Los Angeles' }
        ];
      }
      return [];
    },
    dependsOn: ['country'] // Update when country changes
  }
}

Lifecycle Hooks

{
  controlType: 'input',
  options: {
    formControlName: 'name',
    label: 'Name',
    // Called when field initializes
    onInit: (form, field) => {
      console.log('Field initialized');
    },
    // Called when field is destroyed
    onDestroy: (form, field) => {
      console.log('Field destroyed');
    },
    // Called when any field changes
    onFieldChange: (fieldName, value, form, field) => {
      console.log(${fieldName} changed to:, value);
    }
  }
}
