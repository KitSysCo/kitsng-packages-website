Lookup Select Field

The lookup-select field type provides a dropdown that loads options from an API with pagination, filtering, and multiple data provider options.

Basic Usage

Using Generic Data Provider (Recommended)

import { ILookupDataProvider, LookupDataLoadConfig, LookupResponse } from 'kitsng-form-factory';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

class MyDataProvider implements ILookupDataProvider {
  constructor(private http: HttpClient) {}

loadData(config: LookupDataLoadConfig): Observable {
    return this.http.post('/api/search', {
      page: config.pageIndex,
      size: config.pageSize,
      search: config.search
    }).pipe(
      map(response =&gt; ({
        items: response.data.results,
        totalItems: response.data.total
      }))
    );
  }

    return this.http.get(/api/items/${id}

Using Load Function

{
  controlType: 'lookup-select',
  colSize: 'col-span-12',
  options: {
    formControlName: 'customer',
    label: 'Customer',
    placeholder: 'Search customers...',
    loadData: (config) =&gt; {
      return this.customerService.search({
        page: config.pageIndex,
        size: config.pageSize,
        query: config.search
      });
    },
    extractItems: (response) =&gt; response.results,
    extractTotalItems: (response) =&gt; response.total,
    optionLabel: 'fullName',
    optionValue: 'id',
    pageSize: 25,
    validators: {
      required: true
    }
  }
}

Options

Common Options

Option Type Default Description
formControlName string Required Name of the form control
label string - Label displayed above the field
placeholder string - Placeholder text
dataProvider ILookupDataProvider - Generic data provider (preferred)
loadData function - Custom load function
extractItems function - Function to extract items from response
extractTotalItems function - Function to extract total count
entity string - Entity name (legacy API)
module string - Module name (legacy API)
version string - API version (legacy API)
apiBaseUrl string - Base URL for API
optionLabel string 'text' Property name for option label
optionValue string 'id' Property name for option value
pageSize number 50 Number of items per page
filter boolean true Enable search/filter
showClear boolean false Show clear button
memoryCache boolean false Cache loaded data in memory
forceGetOptionById boolean false Force load option by ID when setting value

Basic Lookup Select

  controlType: 'lookup-select',
  colSize: 'col-span-12',
  options: {

    filter: true
  }

}

Custom Data Provider Example

import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { ILookupDataProvider, LookupDataLoadConfig, LookupResponse } from 'kitsng-form-factory';

@Injectable({ providedIn: 'root' })
export class ProductDataProvider implements ILookupDataProvider {
  constructor(private http: HttpClient) {}

loadData(config: LookupDataLoadConfig): Observable {
    const params = {
      page: config.pageIndex,
      size: config.pageSize,

      ...config.query
    };

return this.http.post('/api/products/search', params).pipe(
      map((response: any) =&gt; ({
        items: response.data.products,
        totalItems: response.data.totalCount
      }))
    );
  }

    return this.http.get(/api/products/${id}
