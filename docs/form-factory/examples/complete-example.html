Control Layout Examples

Vertical Layout (Default)

{
  controlType: 'input',
  options: {
    formControlName: 'field1',
    label: 'Field Label',
    controlLayout: 'vertical' // Label above control
  }
}

Horizontal Layout with Custom Columns

{
  controlType: 'input',
  options: {
    formControlName: 'field2',
    label: 'Field Label',
    controlLayout: 'horizontal', // Label and control side-by-side
    labelColSize: 'col-span-3',  // Label takes 3 columns
    controlColSize: 'col-span-9' // Control takes 9 columns
  }
}

Horizontal Layout with Custom Container

{
  controlType: 'input',
  options: {
    formControlName: 'field3',
    label: 'Field Label',
    controlLayout: 'horizontal',
    labelColSize: 'col-span-4',
    controlColSize: 'col-span-8',
    horizontalContainerClass: 'kitsng-grid kitsng-row kitsng-align-items-center'
  }
}
