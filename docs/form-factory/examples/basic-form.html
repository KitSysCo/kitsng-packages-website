Basic Form Example

This example demonstrates a simple user registration form with basic validation.

Complete Example

import { Component, OnInit } from '@angular/core';
import { FormGroup } from '@angular/forms';
import { KitsngFormFactoryModel } from 'kitsng-form-factory';


@Component({
  selector: 'app-user-registration',
  template: 
    

      
User Registration

      
      
      
      
      

        Please correct the errors above.
      

      
      

        Registration successful!
      

    

  ,
  styles: [
    .container {
      max-width: 600px;
      margin: 0 auto;
      padding: 20px;
    }
  ]
})
export class UserRegistrationComponent implements OnInit {
  userForm = new FormGroup({});
  submitted = false;
  
  formFields: KitsngFormFactoryModel[] = [
    {
      controlType: 'input',
      colSize: 'col-span-12',
      options: {
        formControlName: 'firstName',
        label: 'First Name',
        placeholder: 'Enter your first name',
        validators: {
          required: true,
          minLength: 2,
          maxLength: 50
        },
        errorMessage: 'First name is required (2-50 characters)'
      }
    },
    {
      controlType: 'input',
      colSize: 'col-span-12',
      options: {
        formControlName: 'lastName',
        label: 'Last Name',
        placeholder: 'Enter your last name',
        validators: {
          required: true,
          minLength: 2,
          maxLength: 50
        },
        errorMessage: 'Last name is required (2-50 characters)'
      }
    },
    {
      controlType: 'input',
      colSize: 'col-span-12',
      options: {
        formControlName: 'email',
        label: 'Email Address',
        placeholder: 'Enter your email',
        type: 'text',
        autocomplete: 'email',
        validators: {
          required: true,
          email: true
        },
        errorMessage: 'Please enter a valid email address'
      }
    },
    {
      controlType: 'input',
      colSize: 'col-span-12',
      options: {
        formControlName: 'password',
        label: 'Password',
        placeholder: 'Enter password',
        type: 'password',
        validators: {
          required: true,
          minLength: 8,
          pattern: '^(?=.[a-z])(?=.[A-Z])(?=.*\\d).+$'
        },
        errorMessage: 'Password must be at least 8 characters with uppercase, lowercase, and number'
      }
    },
    {
      controlType: 'dropdown',
      colSize: 'col-span-12',
      options: {
        formControlName: 'country',
        label: 'Country',
        placeholder: 'Select your country',
        dropdownOptions: [
          { id: 'us', text: 'United States' },
          { id: 'ca', text: 'Canada' },
          { id: 'uk', text: 'United Kingdom' },
          { id: 'au', text: 'Australia' }
        ],
        optionLabel: 'text',
        optionValue: 'id',
        validators: {
          required: true
        }
      }
    },
    {
      controlType: 'textarea',
      colSize: 'col-span-12',
      options: {
        formControlName: 'bio',
        label: 'Biography',
        placeholder: 'Tell us about yourself (optional)',
        rows: 5,
        validators: {
          maxLength: 500
        }
      }
    },
    {
      controlType: 'checkbox',
      colSize: 'col-span-12',
      options: {
        formControlName: 'agreeToTerms',
        label: 'I agree to the Terms and Conditions',
        binary: true,
        validators: {
          requiredTrue: true
        },
        errorMessage: 'You must agree to the terms'
      }
    }
  ];



ngOnInit(): void {
    // Initialize form with default values if needed
    this.userForm.patchValue({
      country: 'us'
    });
  }



onSubmit(): void {
    this.submitted = true;
    
    if (this.userForm.valid) {
      console.log('Form submitted:', this.userForm.value);
      
      // Here you would typically call a service to save the data
      // this.userService.register(this.userForm.value).subscribe(...);
      
      // For demo purposes, we'll just log the values
      const formData = {
        firstName: this.userForm.get('firstName')?.value,
        lastName: this.userForm.get('lastName')?.value,
        email: this.userForm.get('email')?.value,
        password: this.userForm.get('password')?.value,
        country: this.userForm.get('country')?.value,
        bio: this.userForm.get('bio')?.value,
        agreeToTerms: this.userForm.get('agreeToTerms')?.value
      };
      
      console.log('Registration data:', formData);
    } else {
      console.log('Form has errors:', this.userForm.errors);
      this.userForm.markAllAsTouched();
    }
  }



onFieldChange(event: { fieldName: string; value: any; form: FormGroup }): void {
    console.log(Field ${event.fieldName} changed to:, event.value);
  }
}

Features Demonstrated

  • ✅ Basic input fields with validation
  • ✅ Email validation
  • ✅ Password validation with pattern
  • ✅ Dropdown selection
  • ✅ Textarea for longer text
  • ✅ Checkbox with requiredTrue validation
  • ✅ Form submission handling
  • ✅ Error display
  • ✅ Field change listeners

Form Structure

The form collects:

  1. First Name - Required, 2-50 characters
  2. Last Name - Required, 2-50 characters
  3. Email - Required, valid email format
  4. Password - Required, minimum 8 chars with uppercase, lowercase, and number
  5. Country - Required dropdown selection
  6. Biography - Optional, max 500 characters
  7. Terms Agreement - Required checkbox
