API Integration

Guide to integrating tables with API endpoints using the generic fetch system.

Overview

The KitsngFetchTableFactory component supports two approaches:

  1. Generic Fetch Function (Recommended) - Works with any API structure
  2. Legacy ApiService - Backward compatible with existing ApiService

Generic Fetch Function (Recommended)

Basic Setup

import { HttpClient } from '@angular/common/http';
import { KitsngTableConfig, KitsngTableQuery, KitsngPaginationResponse } from 'kitsng-table-factory';


@Component({
  template: 
})
export class UsersComponent {
  tableConfig: KitsngTableConfig = {
    columns: [
      { field: 'id', header: 'ID' },
      { field: 'name', header: 'Name' }
    ],
    data: [],
    fetchApiInfo: {
      fetchFunction: (query: KitsngTableQuery) => {
        return this.http.post('/api/users/page', query);
      }
    },
    pageSize: 10
  };



constructor(private http: HttpClient) {}
}

Query Structure

The KitsngTableQuery interface includes:

interface KitsngTableQuery {
  pageIndex?: number;        // Current page (1-based)
  pageSize?: number;         // Items per page
  search?: string;           // Search term
  filters?: any;             // Filter object
  orders?: {                 // Sort orders
















  };
  sortField?: string;        // Sort field
















  [key: string]: any;        // Additional custom params
}

Response Mapping

Standard Response Format

If your API returns the standard format:

{
  items: [...],
  totalItems: 100,
  pageInfo: {
    totalItems: 100,
    pageIndex: 1,
    pageSize: 10
  }
}

No mapping needed:

fetchFunction: (query) => this.http.post('/api/users/page', query)

Custom Response Format

If your API returns a different format, use responseMapper:

fetchApiInfo: {
  fetchFunction: (query) => this.http.post('/api/users/page', query),
  responseMapper: (response: any) => {
    return {
      items: response.data.items,
      totalItems: response.data.totalCount,
      pageInfo: {
        totalItems: response.data.totalCount,
        pageIndex: response.data.page,
        pageSize: response.data.limit
      }
    };
  }
}

Common Response Formats

Format 1: Nested data 
responseMapper: (response) => ({
  items: response.result.data,
  totalItems: response.result.total
})
Format 2: Flat structure 
responseMapper: (response) => ({
  items: response.content,
  totalItems: response.totalElements,
  pageInfo: {
    totalItems: response.totalElements,
    pageIndex: response.number + 1,
    pageSize: response.size
  }
})

Query Building

Transform table queries to your API format:

fetchApiInfo: {
  fetchFunction: (query) => this.http.post('/api/users/page', query),
  queryBuilder: (tableQuery: KitsngTableQuery) => {
    return {
      page: tableQuery.pageIndex,
      limit: tableQuery.pageSize,
      search: tableQuery.search,
      filters: this.transformFilters(tableQuery.filters),
      sortBy: this.getSortField(tableQuery.orders),
      sortOrder: this.getSortOrder(tableQuery.orders)
    };
  }
}

Complete Example

import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Subject } from 'rxjs';
import { KitsngTableConfig, KitsngTableQuery, KitsngPaginationResponse } from 'kitsng-table-factory';


@Component({
  selector: 'app-users',
  template: 
    
  
})
export class UsersComponent implements OnInit {
  tableConfig!: KitsngTableConfig;
  refreshTrigger = new Subject();



constructor(private http: HttpClient) {}



ngOnInit() {
    this.tableConfig = {
      columns: [
        { field: 'id', header: 'ID', sortable: true },
        { field: 'name', header: 'Name', sortable: true },
        { field: 'email', header: 'Email' }
      ],
      data: [],
      fetchApiInfo: {
        fetchFunction: (query: KitsngTableQuery) => {
          const apiQuery = this.buildApiQuery(query);
          return this.http.post('/api/users/page', apiQuery);
        },
        responseMapper: (response: any) => {
          return {
            items: response.data.items,
            totalItems: response.data.totalCount,
            pageInfo: {
              totalItems: response.data.totalCount,
              pageIndex: response.data.page,
              pageSize: response.data.limit
            }
          };
        },
        refreshTrigger: this.refreshTrigger,
        onDataLoaded: (data, pageInfo) => {
          console.log(Loaded ${data.length} of ${pageInfo.totalItems} items);
        },
        onFetchDataError: (error) => {
          console.error('API Error:', error);
          this.handleError(error);
        }
      },
      pageSize: 10,
      pageIndex: 1
    };
  }



private buildApiQuery(query: KitsngTableQuery): any {
    return {

      limit: query.pageSize,
      search: query.search,

Refresh Trigger

Manually trigger data refresh:

export class UsersComponent {
  refreshTrigger = new Subject();


tableConfig: KitsngTableConfig = {
    fetchApiInfo: {
      fetchFunction: (query) => this.http.post('/api/users/page', query),
      refreshTrigger: this.refreshTrigger
    }
  };



refreshData() {
    this.refreshTrigger.next();
  }



// After updating filters
  updateFilters() {
    this.tableConfig.fetchApiInfo!.filters = { status: 'active' };
    this.refreshData();
  }
}

Error Handling

fetchApiInfo: {
  fetchFunction: (query) => this.http.post('/api/users/page', query),
  onFetchDataError: (error) => {
    console.error('API Error:', error);
    
    if (error.status === 404) {
      this.showMessage('Resource not found');
    } else if (error.status === 500) {
      this.showMessage('Server error occurred');
    } else {
      this.showMessage('Failed to load data');
    }
  }
}

Loading State

Loading state is automatically managed:

// Loading state is set automatically during fetch
tableConfig.loading = true;  // Set by component
tableConfig.loadingMessage = 'Loading users...';  // Optional custom message

Search Integration

// Add search to query
fetchApiInfo: {
  fetchFunction: (query) => {
    // query.search contains the search term
    return this.http.post('/api/users/page', {
      ...query,
      search: query.search
    });
  }
}


// Update search term
this.tableConfig.fetchApiInfo!.searchWord = 'john';
this.refreshTrigger.next();

Filter Integration

// Add filters
fetchApiInfo: {
  fetchFunction: (query) => {
    return this.http.post('/api/users/page', {
      ...query,
      filters: query.filters
    });
  },
  filters: {
    status: { value: 'active', level: 0, conOp: 0 }
  }
}


// Update filters
this.tableConfig.fetchApiInfo!.filters = {
  status: { value: 'inactive', level: 0, conOp: 0 },
  role: { value: 'admin', level: 0, conOp: 0 }
};
this.refreshTrigger.next();
