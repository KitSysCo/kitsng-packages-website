KitsngTreeTableFactory Component

Component for displaying hierarchical tree data in a table format.

Usage

Basic Example with Static Data

import { Component } from '@angular/core';
import { KitsngTableConfig } from 'kitsng-table-factory';


@Component({
  selector: 'app-categories',
  template: 
    
  
})
export class CategoriesComponent {
  tableConfig: KitsngTableConfig = {
    columns: [
      { field: 'name', header: 'Category Name' },
      { field: 'description', header: 'Description' }
    ],
    data: [
      {
        id: 1,
        name: 'Electronics',
        description: 'Electronic items',
        items: [
          {
            id: 11,
            name: 'Phones',
            description: 'Mobile phones',
            items: [
              { id: 111, name: 'iPhone', description: 'Apple iPhone' },
              { id: 112, name: 'Samsung', description: 'Samsung Galaxy' }
            ]
          },
          {
            id: 12,
            name: 'Laptops',
            description: 'Laptop computers',
            items: []
          }
        ]
      }
    ],
    treeChildrenKeyName: 'items'  // Key name for children array
  };
}

Tree Table with API Data

import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { KitsngTableConfig, KitsngTableQuery, KitsngPaginationResponse } from 'kitsng-table-factory';
import { Subject } from 'rxjs';


@Component({
  selector: 'app-categories',
  template: 
    
  
})
export class CategoriesComponent implements OnInit {
  tableConfig!: KitsngTableConfig;
  refreshTrigger = new Subject();



constructor(private http: HttpClient) {}



ngOnInit() {
    this.tableConfig = {
      columns: [
        { field: 'name', header: 'Category Name' },
        { field: 'description', header: 'Description' }
      ],
      data: [],
      fetchApiInfo: {
        fetchFunction: (query: KitsngTableQuery) => {
          return this.http.post('/api/categories/page', query);
        },
        refreshTrigger: this.refreshTrigger
      },
      treeChildrenKeyName: 'items',
      treeLazyLoad: true,  // Enable lazy loading of children
      treeShowLoader: true
    };
  }
}

Configuration Options

Tree Structure

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  treeChildrenKeyName: 'items',  // Default: 'items'
  // The key name in your data object that contains children
};

Lazy Loading

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) => this.http.post('/api/categories/page', query)
  },
  treeLazyLoad: true,  // Load children on demand
  treeShowLoader: true,  // Show loading indicator
  treeNodeChildrenFilterKey: 'parentId',  // Filter key for fetching children
  treeNodeChildrenFilterValue: 'id',  // Value field from parent node
  treeNodeChildrenFilters: {}  // Additional filters for children
};

Styling

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  treeClass: 'rounded-xl',
  treeStyle: { 'min-width': '50rem' }
};

Lazy Loading Children

When a node is expanded, the component can fetch its children:

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) => this.http.post('/api/categories/page', query)
  },
  treeLazyLoad: true,
  treeNodeChildrenFilterKey: 'parentId',
  treeNodeChildrenFilterValue: 'id',
  treeNodeChildrenFilters: {
    type: { value: 'category', level: 0, conOp: 0 }
  }
};

When a node is expanded:

  • The component fetches children using the parent node's ID
  • Uses treeNodeChildrenFilterKey as the filter field
  • Uses treeNodeChildrenFilterValue from the parent node as the filter value

Row Selection

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  showSelectionCheckbox: true,
  selectionMode: 'multiple',
  onSelectedItems: (selectedItems) => {
    console.log('Selected nodes:', selectedItems);
  }
};

    console.log('Sort:', event);
  }

import { HttpClient } from '@angular/common/http';

export class CategoriesComponent implements OnInit {
  tableConfig!: KitsngTableConfig;
  refreshTrigger = new Subject();



constructor(private http: HttpClient) {}



ngOnInit() {
    this.tableConfig = {
      columns: [
        { field: 'name', header: 'Category Name', sortable: true },
        { field: 'description', header: 'Description' },
        { field: 'count', header: 'Items Count' }
      ],
      data: [],
      showSelectionCheckbox: true,
      actionButtons: [
        {
          icon: 'pi pi-pencil',
          label: 'Edit',
          onClick: (item) => this.editCategory(item)
        },
        {
          icon: 'pi pi-trash',
          label: 'Delete',
          colorClass: 'p-button-danger',
          onClick: (item) => this.deleteCategory(item)
        }
      ],
      fetchApiInfo: {
        fetchFunction: (query: KitsngTableQuery) => {
          return this.http.post('/api/categories/page', query);
        },
        refreshTrigger: this.refreshTrigger,
        onDataLoaded: (data, pageInfo) => {
          console.log('Categories loaded:', data.length);
        }
      },
      treeChildrenKeyName: 'items',
      treeLazyLoad: true,
      treeShowLoader: true,
      treeNodeChildrenFilterKey: 'parentId',
      treeNodeChildrenFilterValue: 'id',
      onSorting: (event) => {
        console.log('Sorting tree:', event);
      }
    };
  }



editCategory(item: any) {
    console.log('Edit category:', item);
  }



deleteCategory(item: any) {
    console.log('Delete category:', item);
  }
}

Data Structure

Your data should be structured with a children array:

interface TreeNode {
  id: number;
  name: string;
  items?: TreeNode[];  // Children array
}

Or with a custom key name:

interface CategoryNode {
  id: number;
  name: string;
  children?: CategoryNode[];  // Custom key name
}


// Configure with:
treeChildrenKeyName: 'children'
