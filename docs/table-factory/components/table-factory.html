KitsngTableFactory Component

The basic table component for displaying static data or data passed directly.

Usage

Basic Example

import { Component } from '@angular/core';
import { KitsngTableConfig } from 'kitsng-table-factory';


@Component({
  selector: 'app-users',
  template: 
    
  
})
export class UsersComponent {
  tableConfig: KitsngTableConfig = {
    columns: [
      { field: 'id', header: 'ID' },
      { field: 'name', header: 'Name' },
      { field: 'email', header: 'Email' }
    ],
    data: [
      { id: 1, name: 'John Doe', email: 'john@example.com' },
      { id: 2, name: 'Jane Smith', email: 'jane@example.com' }
    ]
  };
}

Inputs

Property Type Required Description
tableConfig KitsngTableConfig Yes Complete table configuration

Features

Column Configuration

columns: [
  {
    field: 'id',
    header: 'ID',
    sortable: true,
    width: '100px',
    visible: true
  },
  {
    field: 'name',
    header: 'Name',
    sortable: true,
    headerCellClass: 'text-bold',
    bodyCellClass: 'text-primary'
  }
]

Sorting

tableConfig: KitsngTableConfig = {
  columns: [
    { field: 'name', header: 'Name', sortable: true }
  ],
  data: [...],
  onSorting: (event) => {
    console.log('Sort field:', event.field);
    console.log('Sort order:', event.order);
    // Handle sorting
  }
};

Row Selection

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  showSelectionCheckbox: true,
  selectionMode: 'multiple', // 'single' or 'multiple'
  selectedItems: [],
  onSelectedItems: (selectedItems) => {
    console.log('Selected:', selectedItems);
  }
};

Row Actions

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  actionButtons: [
    {
      icon: 'pi pi-pencil',
      label: 'Edit',
      onClick: (item) => this.edit(item)
    },
    {
      icon: 'pi pi-trash',
      label: 'Delete',
      colorClass: 'p-button-danger',
      onClick: (item) => this.delete(item)
    }
  ]
};

Row Click Events

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  onRowClick: (event, rowData) => {
    console.log('Row clicked:', rowData);
    this.navigateToDetail(rowData);
  },
  onRowDblClick: (event, rowData) => {
    console.log('Row double-clicked:', rowData);
    this.openQuickView(rowData);
  },
  onCellClick: (event, rowData, column) => {
    console.log('Cell clicked:', column.field, rowData);
  }
};

Expandable Rows

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  expandableRows: true,
  expandedRows: [],
  rowExpandTemplate: this.detailsTemplate,
  onRowExpand: (event) => {
    console.log('Row expanded:', event.data);
  },
  onRowCollapse: (event) => {
    console.log('Row collapsed:', event.data);
  }
};

Conditional Row Styling

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  rowClass: (rowData) => {
    if (rowData.status === 'active') return 'kitsng-bg-blue-50';
    if (rowData.status === 'inactive') return 'kitsng-bg-gray-100';
    return '';
  },
  rowStyle: (rowData) => {
    if (rowData.highlight) {
      return { backgroundColor: '#ffff00' };
    }
    return {};
  }
};

Column Visibility Toggle

tableConfig: KitsngTableConfig = {
  columns: [
    { field: 'id', header: 'ID', visible: true },
    { field: 'name', header: 'Name', visible: true },
    { field: 'email', header: 'Email', visible: false }
  ],
  data: [...],
  showColumnToggle: true  // Shows column visibility toggle button
};

Pagination

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  showPaginator: true,
  showClientPaginator: true,  // Client-side pagination
  pageSize: 10,
  rowsPerPageOptions: [5, 10, 25, 50, 100],
  onPageChanged: (event) => {
    console.log('Page:', event.page);
    console.log('Rows:', event.rows);
  }
};

Loading State

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [...],
  loading: true,
  loadingMessage: 'Loading data...'
};

Empty State

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  emptymessage: 'No users found',
  emptymessageTemplate: this.emptyTemplate  // Optional custom template
};

Complete Example

import { Component, ViewChild, TemplateRef } from '@angular/core';
import { KitsngTableConfig } from 'kitsng-table-factory';


@Component({
  selector: 'app-users',
  template: 
    
      
        
















        
      
    
  
})
export class UsersComponent {
  tableConfig: KitsngTableConfig = {
    columns: [
      { field: 'id', header: 'ID', sortable: true, width: '100px' },
      { field: 'name', header: 'Name', sortable: true },
      { field: 'email', header: 'Email' },
      { field: 'status', header: 'Status' }
    ],
    data: [
      { id: 1, name: 'John Doe', email: 'john@example.com', status: 'active' },
      { id: 2, name: 'Jane Smith', email: 'jane@example.com', status: 'inactive' }
    ],
    showSelectionCheckbox: true,
    selectionMode: 'multiple',
    actionButtons: [
      {
        icon: 'pi pi-pencil',
        label: 'Edit',
        onClick: (item) => this.edit(item)
      },
      {
        icon: 'pi pi-trash',
        label: 'Delete',
        colorClass: 'p-button-danger',
        onClick: (item) => this.delete(item)
      }
    ],
    showColumnToggle: true,
    onRowClick: (event, rowData) => {
      this.viewDetails(rowData);
    }
  };



edit(item: any) {
    console.log('Edit:', item);
  }



delete(item: any) {
    console.log('Delete:', item);
  }



viewDetails(item: any) {
    console.log('View details:', item);
  }
}
