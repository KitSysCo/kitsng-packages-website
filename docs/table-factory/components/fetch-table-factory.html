KitsngFetchTableFactory Component

Table component that automatically fetches data from an API endpoint.

Usage

Basic Example

import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { KitsngTableConfig, KitsngTableQuery, KitsngPaginationResponse } from 'kitsng-table-factory';
import { Observable } from 'rxjs';


@Component({
  selector: 'app-users',
  template: 
    
  
})
export class UsersComponent implements OnInit {
  tableConfig!: KitsngTableConfig;



constructor(private http: HttpClient) {}



ngOnInit() {
    this.tableConfig = {
      columns: [
        { field: 'id', header: 'ID', sortable: true },
        { field: 'name', header: 'Name', sortable: true },
        { field: 'email', header: 'Email' }
      ],
      data: [],
      fetchApiInfo: {
        fetchFunction: (query: KitsngTableQuery) => {
          return this.http.post('/api/users/page', query);
        }
      },
      pageSize: 10,
      pageIndex: 1
    };
  }
}

Generic API Fetch (Recommended)

Simple Example

fetchApiInfo: {
  fetchFunction: (query: KitsngTableQuery) => {
    return this.http.post('/api/users/page', query);
  }
}

With Query Builder

fetchApiInfo: {
  fetchFunction: (query: KitsngTableQuery) => {
    return this.http.post('/api/users/page', query);
  },
  queryBuilder: (query: KitsngTableQuery) => {
    // Transform table query to your API format
    return {
      page: query.pageIndex,
      limit: query.pageSize,
      search: query.search,
      sort: this.buildSortQuery(query.orders)
    };
  }
}

With Response Mapper

fetchApiInfo: {
  fetchFunction: (query: KitsngTableQuery) => {
    return this.http.post('/api/users/page', query);
  },
  responseMapper: (response: any) => {
    // Transform API response to standard format
    return {
      items: response.data.items,
      totalItems: response.data.totalCount,
      pageInfo: {
        totalItems: response.data.totalCount,
        pageIndex: response.data.page,
        pageSize: response.data.limit
      }
    };
  }
}

Loading State

The component automatically manages loading state:

// Loading state is automatically set during fetch
tableConfig.loading = true;  // Set automatically
tableConfig.loadingMessage = 'Loading users...';  // Optional custom message

Pagination

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) => this.http.post('/api/users/page', query)
  },
  pageSize: 10,
  pageIndex: 1,
  showPaginator: true,
  onPageChanged: (event) => {
    this.tableConfig.pageIndex = event.page + 1;
    this.tableConfig.pageSize = event.rows;
    // Trigger refresh if needed
    this.refreshTrigger.next();
  }
};

Sorting

tableConfig: KitsngTableConfig = {
  columns: [
    { field: 'name', header: 'Name', sortable: true }
  ],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) => this.http.post('/api/users/page', query)
  },
  onSorting: (event) => {
    // Custom sorting handler
    console.log('Sort:', event.field, event.order);
  }
  // If onSorting is not provided, sorting is handled automatically
};

Search

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) => this.http.post('/api/users/page', query),
    queryBuilder: (query) => {
      return {
        ...query,
        search: query.search  // Search term is included automatically
      };
    }
  }
};


// Update search term
this.tableConfig.fetchApiInfo!.searchWord = 'john';
this.refreshTrigger.next();

Filtering

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) => this.http.post('/api/users/page', query),
    filters: {
      status: { value: 'active', level: 0, conOp: 0 }
    }
  }
};


// Update filters
this.tableConfig.fetchApiInfo!.filters = {
  status: { value: 'inactive', level: 0, conOp: 0 }
};
this.refreshTrigger.next();

Manual Refresh

export class UsersComponent {
  refreshTrigger = new Subject();


tableConfig: KitsngTableConfig = {
    fetchApiInfo: {
      fetchFunction: (query) => this.http.post('/api/users/page', query),
      refreshTrigger: this.refreshTrigger
    }
  };



refreshData() {
    this.refreshTrigger.next();
  }
}

Error Handling

  onFetchDataError: (error) => {
    console.error('API Error:', error);
    this.showErrorMessage('Failed to load data');
    // Handle error (show toast, log, etc.)
  }
}

Complete Example with All Features

import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Subject } from 'rxjs';
import { KitsngTableConfig, KitsngTableQuery, KitsngPaginationResponse } from 'kitsng-table-factory';


@Component({
  selector: 'app-users',
  template: 
    
      
        {{ value }}
      
    
  
})
export class UsersComponent implements OnInit {
  tableConfig!: KitsngTableConfig;
  refreshTrigger = new Subject();



constructor(private http: HttpClient) {}



ngOnInit() {
    this.tableConfig = {
      columns: [
        { field: 'id', header: 'ID', sortable: true, width: '100px' },
        { field: 'name', header: 'Name', sortable: true },
        { field: 'email', header: 'Email' },
        { field: 'status', header: 'Status' }
      ],
      data: [],
      showSelectionCheckbox: true,
      actionButtons: [
        {
          icon: 'pi pi-pencil',
          label: 'Edit',
          onClick: (item) => this.edit(item)
        }
      ],
      fetchApiInfo: {
        fetchFunction: (query: KitsngTableQuery) => {
          return this.http.post('/api/users/page', query);
        },
        queryBuilder: (query) => ({
          page: query.pageIndex,
          limit: query.pageSize,
          search: query.search,
          filters: query.filters,
          sort: this.buildSort(query.orders)
        }),
        responseMapper: (response) => ({
          items: response.data.items,
          totalItems: response.data.totalCount,
          pageInfo: {
            totalItems: response.data.totalCount,
            pageIndex: response.data.page,
            pageSize: response.data.limit
          }
        }),
        refreshTrigger: this.refreshTrigger,
        onDataLoaded: (data, pageInfo) => {
          console.log(Loaded ${data.length} of ${pageInfo.totalItems} items);
        },
        onFetchDataError: (error) => {
          this.handleError(error);
        }
      },
      pageSize: 10,
      pageIndex: 1,
      onPageChanged: (event) => {
        this.tableConfig.pageIndex = event.page + 1;
        this.tableConfig.pageSize = event.rows;
      }
    };
  }



refresh() {
    this.refreshTrigger.next();
  }



edit(item: any) {
    console.log('Edit:', item);
  }



private buildSort(orders: any): any {

















    const field = Object.keys(orders)[0];
    const order = orders[field];
    return { field, order: order === 0 ? 'asc' : 'desc' };
  }



private handleError(error: any) {
    console.error('Error:', error);
  }
}
