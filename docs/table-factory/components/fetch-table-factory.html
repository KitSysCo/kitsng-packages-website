KitsngFetchTableFactory Component

Table component that automatically fetches data from an API endpoint.

Usage

<kitsng-fetch-table-factory [tableConfig]="tableConfig"></kitsng-fetch-table-factory>

import { Observable } from 'rxjs';

Generic API Fetch (Recommended)

Simple Example

fetchApiInfo: {
  fetchFunction: (query: KitsngTableQuery) =&gt; {
    return this.http.post('/api/users/page', query);
  }
}

With Query Builder

fetchApiInfo: {
  fetchFunction: (query: KitsngTableQuery) =&gt; {
    return this.http.post('/api/users/page', query);
  },
  queryBuilder: (query: KitsngTableQuery) =&gt; {
    // Transform table query to your API format
    return {
      page: query.pageIndex,
      limit: query.pageSize,
      search: query.search,
      sort: this.buildSortQuery(query.orders)
    };
  }
}

With Response Mapper

fetchApiInfo: {
  fetchFunction: (query: KitsngTableQuery) =&gt; {
    return this.http.post('/api/users/page', query);
  },
  responseMapper: (response: any) =&gt; {
    // Transform API response to standard format
    return {
      items: response.data.items,
      totalItems: response.data.totalCount,
      pageInfo: {
        totalItems: response.data.totalCount,
        pageIndex: response.data.page,
        pageSize: response.data.limit
      }
    };
  }
}

Legacy ApiService (Backward Compatible)

If you're using the legacy ApiService:

import { ApiService } from 'kitsng-table-factory';

// In your module providers
providers: [ApiService]

// In your component
fetchApiInfo: {
  module: 'users',
  entity: 'user',
  baseUri: 'https://api.example.com',
  version: 'v1',
  path: 'page'
}

Loading State

The component automatically manages loading state:

// Loading state is automatically set during fetch
tableConfig.loading = true;  // Set automatically
tableConfig.loadingMessage = 'Loading users...';  // Optional custom message

Pagination

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) =&gt; this.http.post('/api/users/page', query)
  },
  pageSize: 10,
  pageIndex: 1,
  showPaginator: true,
  onPageChanged: (event) =&gt; {
    this.tableConfig.pageIndex = event.page + 1;
    this.tableConfig.pageSize = event.rows;
    // Trigger refresh if needed
    this.refreshTrigger.next();
  }
};

Sorting

tableConfig: KitsngTableConfig = {
  columns: [
    { field: 'name', header: 'Name', sortable: true }
  ],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) =&gt; this.http.post('/api/users/page', query)
  },
  onSorting: (event) =&gt; {
    // Custom sorting handler
    console.log('Sort:', event.field, event.order);
  }
  // If onSorting is not provided, sorting is handled automatically
};

Search

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) =&gt; this.http.post('/api/users/page', query),
    queryBuilder: (query) =&gt; {
      return {
        ...query,
        search: query.search  // Search term is included automatically
      };
    }
  }
};

// Update search term
this.tableConfig.fetchApiInfo!.searchWord = 'john';
this.refreshTrigger.next();

Filtering

tableConfig: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) =&gt; this.http.post('/api/users/page', query),
    filters: {
      status: { value: 'active', level: 0, conOp: 0 }
    }
  }
};

// Update filters
this.tableConfig.fetchApiInfo!.filters = {
  status: { value: 'inactive', level: 0, conOp: 0 }
};
this.refreshTrigger.next();

Manual Refresh

export class UsersComponent {
  refreshTrigger = new Subject();

tableConfig: KitsngTableConfig = {
    fetchApiInfo: {
      fetchFunction: (query) =&gt; this.http.post('/api/users/page', query),
      refreshTrigger: this.refreshTrigger
    }
  };

refreshData() {
    this.refreshTrigger.next();
  }
}

Error Handling

fetchApiInfo: {
  fetchFunction: (query) =&gt; this.http.post('/api/users/page', query),
  onFetchDataError: (error) =&gt; {
    console.error('API Error:', error);
    this.showErrorMessage('Failed to load data');
    // Handle error (show toast, log, etc.)
  }
}
