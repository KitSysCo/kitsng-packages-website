Get up and running with kitsng-table-factory in minutes!

Basic Table Example

import { Component } from '@angular/core';
import { KitsngTableConfig } from 'kitsng-table-factory';


@Component({
  selector: 'app-users',
  template: 
    
  
})
export class UsersComponent {
  tableConfig: KitsngTableConfig = {
    columns: [

  };
}

Table with API Data Fetching

import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { KitsngTableConfig, KitsngTableQuery, KitsngPaginationResponse } from 'kitsng-table-factory';
import { Observable } from 'rxjs';


@Component({
  selector: 'app-users',
  template: 
    
  
})
export class UsersComponent implements OnInit {
  tableConfig!: KitsngTableConfig;



constructor(private http: HttpClient) {}



ngOnInit() {
    this.tableConfig = {
      columns: [
        { field: 'id', header: 'ID', sortable: true },
        { field: 'name', header: 'Name', sortable: true },
        { field: 'email', header: 'Email' }
      ],
      data: [],
      fetchApiInfo: {
        fetchFunction: (query: KitsngTableQuery) => {
          return this.http.post('/api/users/page', query);
        },
        responseMapper: (response: any) => ({
          items: response.data.items,
          totalItems: response.data.total
        })
      },
      pageSize: 10,
      pageIndex: 1
    };
  }
}

Complete Example

import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { KitsngTableConfig, KitsngTableQuery, KitsngPaginationResponse } from 'kitsng-table-factory';
import { Observable } from 'rxjs';


@Component({
  selector: 'app-users',
  template: 
    
      
        
          {{ value }}
        
      
    
  
})
export class UsersComponent implements OnInit {
  tableConfig!: KitsngTableConfig;



constructor(private http: HttpClient) {}



ngOnInit() {
    this.tableConfig = {
      columns: [
        { field: 'id', header: 'ID', sortable: true, width: '100px' },
        { field: 'name', header: 'Name', sortable: true },
        { field: 'email', header: 'Email' },
        { field: 'status', header: 'Status' }
      ],
      data: [],
      showSelectionCheckbox: true,
      actionButtons: [
        {
          icon: 'pi pi-pencil',
          label: 'Edit',
          onClick: (item) => this.editUser(item)
        },
        {
          icon: 'pi pi-trash',
          label: 'Delete',
          colorClass: 'p-button-danger',
          onClick: (item) => this.deleteUser(item)
        }
      ],
      bulkActions: [
        {
          icon: 'pi pi-trash',
          label: 'Delete Selected',
          onClick: (selectedItems) => this.deleteSelected(selectedItems)
        }
      ],
      showBulkActions: true,
      fetchApiInfo: {
        fetchFunction: (query: KitsngTableQuery) => {
          return this.http.post('/api/users/page', query);
        }
      },
      pageSize: 10,
      onRowClick: (event, rowData) => {
        console.log('Row clicked:', rowData);
      },
      onSorting: (event) => {
        console.log('Sorting:', event);
      }
    };
  }



editUser(user: any) {
    console.log('Edit user:', user);
  }



deleteUser(user: any) {
    console.log('Delete user:', user);
  }



deleteSelected(users: any[]) {
    console.log('Delete selected:', users);
  }
}
