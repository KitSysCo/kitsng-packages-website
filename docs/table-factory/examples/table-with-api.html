Table with API Example

Example of using KitsngFetchTableFactory to fetch data from an API.

Basic API Example

import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { KitsngTableConfig, KitsngTableQuery, KitsngPaginationResponse } from 'kitsng-table-factory';


@Component({
  selector: 'app-users',
  template: 
    
  
})
export class UsersComponent implements OnInit {
  tableConfig!: KitsngTableConfig;



constructor(private http: HttpClient) {}



ngOnInit() {
    this.tableConfig = {
      columns: [
        { field: 'id', header: 'ID', sortable: true },
        { field: 'name', header: 'Name', sortable: true },
        { field: 'email', header: 'Email' }
      ],
      data: [],
      fetchApiInfo: {
        fetchFunction: (query: KitsngTableQuery) => {
          return this.http.post('/api/users/page', query);
        }
      },
      pageSize: 10,
      pageIndex: 1
    };
  }
}

With Custom Query Builder

fetchApiInfo: {
  fetchFunction: (query: KitsngTableQuery) => {
    const apiQuery = {
      page: query.pageIndex,
      limit: query.pageSize,
      search: query.search,
      sortBy: this.getSortField(query.orders),
      sortOrder: this.getSortOrder(query.orders)
    };
    return this.http.post('/api/users/page', apiQuery);
  },
  queryBuilder: (query) => ({
    page: query.pageIndex,
    limit: query.pageSize,
    search: query.search,
    filters: this.transformFilters(query.filters)
  })
}

With Response Mapper

fetchApiInfo: {
  fetchFunction: (query) => {
    return this.http.post('/api/users/page', query);
  },
  responseMapper: (response: any) => {
    // Transform API response to standard format
    return {
      items: response.data.items,
      totalItems: response.data.totalCount,
      pageInfo: {
        totalItems: response.data.totalCount,
        pageIndex: response.data.page,
        pageSize: response.data.limit
      }
    };
  }
}

With Refresh Trigger

import { Subject } from 'rxjs';


export class UsersComponent {
  refreshTrigger = new Subject();



tableConfig: KitsngTableConfig = {
    fetchApiInfo: {
      fetchFunction: (query) => this.http.post('/api/users/page', query),
      refreshTrigger: this.refreshTrigger
    }
  };



refreshData() {
    this.refreshTrigger.next();
  }



// After creating/updating a user
  onUserSaved() {
    this.refreshData();
  }
}
