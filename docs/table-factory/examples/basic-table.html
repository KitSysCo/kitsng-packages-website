With Sorting

tableConfig: KitsngTableConfig = {
  columns: [
    { field: 'id', header: 'ID', sortable: true },
    { field: 'name', header: 'Name', sortable: true },
    { field: 'email', header: 'Email', sortable: true }
  ],
  data: [
    { id: 1, name: 'John Doe', email: 'john@example.com' },
    { id: 2, name: 'Jane Smith', email: 'jane@example.com' }
  ],
  onSorting: (event) =&gt; {
    console.log('Sort field:', event.field);
    console.log('Sort order:', event.order);
  }
};

With Pagination

tableConfig: KitsngTableConfig = {
  columns: [
    { field: 'id', header: 'ID' },
    { field: 'name', header: 'Name' },
    { field: 'email', header: 'Email' }
  ],
  data: [
    // ... 50 items
  ],
  showPaginator: true,
  showClientPaginator: true,  // Client-side pagination
  pageSize: 10,
  rowsPerPageOptions: [5, 10, 25, 50],
  onPageChanged: (event) =&gt; {
    console.log('Page:', event.page);
    console.log('Rows per page:', event.rows);
  }
};

With Row Selection

tableConfig: KitsngTableConfig = {
  columns: [
    { field: 'id', header: 'ID' },
    { field: 'name', header: 'Name' }
  ],
  data: [
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Smith' }
  ],
  showSelectionCheckbox: true,
  selectionMode: 'multiple',
  selectedItems: [],
  onSelectedItems: (selectedItems) =&gt; {
    console.log('Selected:', selectedItems);
  }
};

With Row Actions

tableConfig: KitsngTableConfig = {
  columns: [
    { field: 'id', header: 'ID' },
    { field: 'name', header: 'Name' }
  ],
  data: [
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Smith' }
  ],
  actionButtons: [
    {
      icon: 'pi pi-pencil',
      label: 'Edit',
      onClick: (item) =&gt; {
        console.log('Edit:', item);
      }
    },
    {
      icon: 'pi pi-trash',
      label: 'Delete',
      colorClass: 'p-button-danger',
      onClick: (item) =&gt; {
        console.log('Delete:', item);
      }
    }
  ]
};
