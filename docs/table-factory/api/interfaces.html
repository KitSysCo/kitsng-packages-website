TypeScript Interfaces

Complete reference for all TypeScript interfaces and types.

KitsngTableConfig

Main configuration interface for all table components.

interface KitsngTableConfig {
  // Required
  columns: KitsngColumnOption[];
  data: any[];


// Selection
  selectedItems?: any[];
  onSelectedItems?(event: any): void;
  showSelectionCheckbox?: boolean;


















// Pagination
  first?: number;
  totalRecords?: number;
  pageSize?: number;
  pageIndex?: number;
  rowsPerPageOptions?: any[];
  showPaginator?: boolean;
  showClientPaginator?: boolean;
  onPageChanged?(event: any): void;
  paginatorClass?: string;



// Sorting
  onSorting?(event: any): void;



// Events
  onRowClick?(event: any, rowData: any): void;
  onCellClick?(event: any, rowData: any, column: KitsngColumnOption): void;
  onRowDblClick?(event: any, rowData: any): void;



// Actions
  actionButtons?: KitsngTabelAction[];
  actionTemplate?: TemplateRef;
  actionCellHeaderStyle?: { [key: string]: string };
  actionCellBodyStyle?: { [key: string]: string };



// Bulk Actions
  bulkActions?: KitsngTabelAction[];
  showBulkActions?: boolean;



// Styling
  rowHover?: boolean;




















  containerClass?: string;
  class?: string;
  tableStyle?: { [key: string]: string };



// Loading
  loading?: boolean;
  loadingMessage?: string;



// Expandable Rows
  expandableRows?: boolean;
















  onRowExpand?(event: any): void;
  onRowCollapse?(event: any): void;
  rowExpandTemplate?: TemplateRef;



// Column Visibility
  showColumnToggle?: boolean;



// Empty State
  emptymessage?: string;
  emptymessageTemplate?: TemplateRef;



// Templates
  headerTemplate?: TemplateRef;
  bodyTemplate?: TemplateRef;
  footerTemplate?: TemplateRef;
  summaryTemplate?: TemplateRef;
  captionTemplate?: TemplateRef;
  tableTemplate?: TemplateRef;



// API
  fetchApiInfo?: KitsngFetchApiInfo;



// Table Options
  resizableColumns?: boolean;
  scrollable?: boolean;
  currentPageReportTemplate?: string;
  trackBy?: (index: number, item: any) => any;



// Tree Table
  treeNodeChildrenFilterKey?: string;
  treeNodeChildrenFilterValue?: string;
  treeNodeChildrenFilters?: any;
  treeShowLoader?: boolean;
  treeLazyLoad?: boolean;
  treeChildrenKeyName?: string;
  treeStyle?: { [key: string]: string };
  treeClass?: string;
}

KitsngColumnOption

Column configuration interface.

interface KitsngColumnOption {
  field: string;                    // Required: Data field name
  header: string;                   // Required: Column header
  sortable?: boolean;               // Enable sorting







'image'
'severity'









  template?: TemplateRef;      // Custom template
  pipes?: KitsngPipeConfig[];      // Pipes to apply
  headerCellClass?: string;         // Header cell CSS class
  headerCellStyle?: { [key: string]: string };
  bodyCellClass?: string;           // Body cell CSS class
  bodyCellStyle?: { [key: string]: string };
  additionalData?: any;             // Additional data
  isFrozen?: boolean;               // Freeze column
















  width?: string;                   // Column width
  minWidth?: string;                // Minimum width
  maxWidth?: string;                // Maximum width
  visible?: boolean;                 // Column visibility
  exportable?: boolean;              // Include in export
  filterable?: boolean;             // Enable filtering







'contains'
'endsWith'
'equals'
'notEquals'
'lt'
'lte'
'gt'









}

KitsngFetchApiInfo

API fetch configuration interface.

interface KitsngFetchApiInfo {
  // Generic Fetch (Recommended)
  fetchFunction?: KitsngFetchFunction;
  responseMapper?: KitsngResponseMapper;
  queryBuilder?: (query: KitsngTableQuery) => any;
  refreshTrigger?: Subject;
  onDataLoaded?: (data: any[], pageInfo: any) => void;
  onFetchDataError?: (error: any) => void;


// Legacy ApiService (Deprecated)
  module?: string;
  entity?: string;
  baseUri?: string;
  version?: string;
  path?: string;
  filters?: any;
  orders?: any;
  searchWord?: any;
  _getData?: Subject;
  queryMapper?: (query: any) => any;
}

KitsngPaginationResponse

Pagination response interface.

interface KitsngPaginationResponse {
  items: T[];
  totalItems?: number;
  totalPages?: number;
  pageIndex?: number;
  pageSize?: number;
  pageInfo?: {
    totalItems: number;
    totalPages?: number;
    pageIndex?: number;
    pageSize?: number;
  };
  page?: {
    totalItems: number;
    totalPages?: number;
    pageIndex?: number;
    pageSize?: number;
  };
}

interface KitsngTableQuery {

  filters?: any;
















  sortField?: string;
















  [key: string]: any;  // Allow additional custom parameters
}

type KitsngFetchFunction = (

Type for response mapper function.

type KitsngResponseMapper = (

Usage Examples

Basic Configuration

const config: KitsngTableConfig = {
  columns: [
    { field: 'id', header: 'ID' },
    { field: 'name', header: 'Name' }
  ],
  data: [
    { id: 1, name: 'John' }
  ]
};

With API

const config: KitsngTableConfig = {
  columns: [...],
  data: [],
  fetchApiInfo: {
    fetchFunction: (query) => this.http.post('/api/data', query)
  }
};
